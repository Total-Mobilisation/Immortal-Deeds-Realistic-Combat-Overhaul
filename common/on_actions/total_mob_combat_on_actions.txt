#Distributes historical CEV values to all relevant nations

on_actions = {
	on_startup = {
		effect = {
			every_country = {
				#Add terrain modifiers
				set_technology = {
					combat_modifiers = 1
					popup = no
				}
			}
		}
	}
	on_daily = {
		effect = {
				#Add terrain modifiers
				set_technology = {
					combat_modifiers = 1
					popup = no
				}
				#Add dynamic modifiers
				add_dynamic_modifier = { modifier = total_mob_combat_offsets }
				add_ideas = totalmob_equipment_modifiers
				#Offsets max_planning_factor
				set_variable = { total_mob_planning_offset = -0.75 }
				#CEV Calculations
				IF = {
					LIMIT = {
						NOT = { has_variable = { total_mob_cev_modifier }}
						original_tag = GER
					}
					set_variable = { total_mob_cev_modifier = 0 }
				}
				IF = {
					LIMIT = {
						NOT = { has_variable = { total_mob_cev_modifier }}
						OR = {
							original_tag = USA
							original_tag = FIN
						}
					}
					set_variable = { total_mob_cev_modifier = -0.1 }
				}
				IF = {
					LIMIT = {
						NOT = { has_variable = { total_mob_cev_modifier }}
						OR = {
							original_tag = ENG
							original_tag = CAN
							original_tag = AST
							original_tag = NZL
						}
					}
					set_variable = { total_mob_cev_modifier = -0.15 }
				}
				IF = {
					LIMIT = {
						NOT = { has_variable = { total_mob_cev_modifier }}
						OR = {
							original_tag = SOV
							original_tag = JAP
							original_tag = AUS
							original_tag = SAF
						}
					}
					set_variable = { total_mob_cev_modifier = -0.2 }
				}
				IF = {
					LIMIT = {
						NOT = { has_variable = { total_mob_cev_modifier }}
						OR = {
							original_tag = RAJ
							original_tag = POL
							original_tag = CZE
							original_tag = SWE
							original_tag = SWI
						}
					}
					set_variable = { total_mob_cev_modifier = -0.25 }
				}
				IF = {
					LIMIT = {
						NOT = { has_variable = { total_mob_cev_modifier }}
						OR = {
							original_tag = FRA
							original_tag = BEL
							original_tag = HOL
							original_tag = HUN
							original_tag = SPR
							original_tag = GRE
							original_tag = NOR
							original_tag = LAT
							original_tag = LIT
							original_tag = EST
						}
					}
					set_variable = { total_mob_cev_modifier = -0.3 }
				}
				IF = {
					LIMIT = {
						NOT = { has_variable = { total_mob_cev_modifier }}
						OR = {
							original_tag = YUG
							original_tag = ROM
							original_tag = BRA
							original_tag = POR
						}
					}
					set_variable = { total_mob_cev_modifier = -0.35 }
				}
				#Default Catch All
				IF = {
					LIMIT = {
						NOT = { has_variable = { total_mob_cev_modifier }}
						set_variable = { total_mob_cev_modifier = -0.4 }
					}
				}
		}
	}
}
